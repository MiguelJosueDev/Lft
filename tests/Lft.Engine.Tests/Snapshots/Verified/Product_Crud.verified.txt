import { ReactiveService } from '@livefree/reactive';
import { GeneratedSDK } from 'sdk/GeneratedSDK';
import { GeneratedConfig } from 'core/config';
const ProductsClient = GeneratedSDK.ProductsClient;

export const ProductsService = new (ReactiveService('products', ProductsClient, {
  recursive: true,
  recurseDepth: 2,
}))(GeneratedConfig);

export const ProductsReactor = ProductsService.reactor('productsReactor', {
  success: (response, services, db) => {
    const { source, args, payload } = response;

    switch (source) {

      case ProductsService.getProduct:
        return db.products.put(payload);

      case ProductsService.createProduct:
        services.alerts.actionSuccess(`Product added successfully`, 'products', response);
        return db.products.add(payload);

      case ProductsService.updateProduct:
        services.alerts.actionSuccess(`Product updated successfully`, 'products', response);
        return db.products.put(payload);

      case ProductsService.deleteProduct:
        services.alerts.actionSuccess(`Product deleted successfully`, 'products', response);
        return db.products.delete(args[0]);
    }
  },
  error: (response, services) => {
    switch (response.source) {

      case ProductsService.getProduct:
        return services.alerts.actionError(`Failed to fetch Product details`, 'products', response);

      case ProductsService.createProduct:
        return services.alerts.actionError(`Failed to create Product`, 'products', response);

      case ProductsService.updateProduct:
        return services.alerts.actionError(`Failed to update Product`, 'products', response);

      case ProductsService.deleteProduct:
        return services.alerts.actionError(`Failed to delete Product`, 'products', response);
    }
  },
});

import { useCallback } from 'react';
import { TableRow, TableColumn, TableCell } from '@heroui/react';;
import { RequestBrokerDataTable } from '@livefree/react-ui';

const columns = [
  { propertyName: 'id', name: 'Id', isSortable: true },
];

export const ProductsList = ({ products = [], filters, onSelect }) => {
  const renderCell = useCallback((row, columnKey) => {
    const cellValue = row[columnKey];

    switch (columnKey) {
      default:
        return cellValue;
    }
  }, []);

  return (
    <RequestBrokerDataTable
      columns={columns}
      items={ products }
      filters={filters}
      renderColumn={column => (
        <TableColumn
          key={column.propertyName}
          allowsSorting={column?.isSortable}
        >
          {column.name}
        </TableColumn>
      )}
      renderRow={row => (
        <TableRow key={row.id} onClick={() => onSelect(row)}>
          {columnKey => <TableCell key={columnKey}>{renderCell(row, columnKey)}</TableCell>}
        </TableRow>
      )}
      withPagination
      withSorting
      withColumnSelection
    />
  );
};

import { useCallback } from 'react';
import { Reactive } from '@livefree/reactive';
import { RequestBroker, queryDexieWithMetadata } from '@livefree/applications';
import { ProductsList } from 'features/products/components';

export const ProductsListController = Reactive(
  ({ data, services, monitors, db, onSelect }) => {
    const { products } = data;

    const handleExecute = useCallback((action, metadata) => {
      action(metadata);
    }, []);

    return (
      <RequestBroker
        actionKey="queryProducts"
        action={services.products.queryProducts}
        defaultSorting={ { propertyName: 'id', sortType: false } }
        defaultPaging={ { skip: 0, take: 50, page: 1, totalCount: 0 } }
        isExecuting={monitors.queryProducts.executing}
        skipInitial={() => db.products.toCollection().count()}
        onExecute={handleExecute}
      >
        <ProductsList
          products={ products }
          isExecuting={monitors.queryProducts.executing}
          onSelect={item => onSelect && onSelect(item)}
        />
      </RequestBroker>
    );
  },
  {
    name: 'ProductsListController',
    queries: (_, services) => [
      {
        name: 'products',
        execute: db =>
          queryDexieWithMetadata(db, services.products.queryProducts.action, () => db.products, {
            sorting: true,
          }),
        deps: [],
      },
    ],
    monitors: services => [services.products.queryProducts],
  },
);

import { useState } from 'react';
import { Reactive, useReactorOnSuccess } from '@livefree/reactive';
import { DeleteButton, SaveButton, withModalController } from '@livefree/react-ui';
import { ProductForm } from 'features/products/components';

const ProductUpdateController = Reactive(
  ({ services, data, monitors, productId, onSuccess, onClose }) => {
    const { product } = data;
    const [form, setForm] = useState({});

    useReactorOnSuccess(services.products, 'updateProduct', payload => {
      onSuccess && onSuccess(payload);
    });

    useReactorOnSuccess(services.products, 'deleteProduct', () => {
      onClose && onClose();
    });

    return (
      <div className="flex flex-col gap-2">
        <ProductForm product={ product } onChange={setForm} />
        <div className="flex justify-between gap-2">
          <DeleteButton
            size="sm"
            onPress={() => services.products.deleteProduct(productId)}
            isLoading={monitors.deleteProduct.executing}
          />
          <SaveButton
            size="sm"
            isDisabled={!form.isValid}
            onPress={() =>
              services.products.updateProduct(productId, form.data)
            }
            isLoading={monitors.updateProduct.executing}
          />
        </div>
      </div>
    );
  },
  {
    name: 'ProductUpdateController',
    sync: (services, { productId }) => [
      {
        execute: () => services.products.getProduct(productId),
        deps: [productId],
      },
    ],
    queries: ({ productId }) => [
      {
        name: 'product',
        defaultValue: {},
        execute: db => db.products.get(productId),
        deps: [productId],
      },
    ],
    monitors: services => [
      services.products.updateProduct,
      services.products.deleteProduct,
    ],
  },
);

ProductUpdateController.Modal = withModalController(({ onSuccess, ...props }) => (
  <ProductUpdateController onSuccess={onSuccess} {...props} />
));

export { ProductUpdateController };
