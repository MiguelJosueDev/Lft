  core/services/CategoriesService.js:
import { ReactiveService } from '@livefree/reactive';
import { GeneratedSDK } from 'sdk/GeneratedSDK';
import { GeneratedConfig } from 'core/config';
const CategoriesClient = GeneratedSDK.CategoriesClient;

export const CategoriesService = new (ReactiveService('categories', CategoriesClient, {
  recursive: true,
  recurseDepth: 2,
}))(GeneratedConfig);

export const CategoriesReactor = CategoriesService.reactor('categoriesReactor', {
  success: (response, services, db) => {
    const { source, args, payload } = response;

    switch (source) {

      case CategoriesService.getCategory:
        return db.categories.put(payload);

      case CategoriesService.createCategory:
        services.alerts.actionSuccess(`Category added successfully`, 'categories', response);
        return db.categories.add(payload);

      case CategoriesService.updateCategory:
        services.alerts.actionSuccess(`Category updated successfully`, 'categories', response);
        return db.categories.put(payload);

      case CategoriesService.deleteCategory:
        services.alerts.actionSuccess(`Category deleted successfully`, 'categories', response);
        return db.categories.delete(args[0]);
    }
  },
  error: (response, services) => {
    switch (response.source) {

      case CategoriesService.getCategory:
        return services.alerts.actionError(`Failed to fetch Category details`, 'categories', response);

      case CategoriesService.createCategory:
        return services.alerts.actionError(`Failed to create Category`, 'categories', response);

      case CategoriesService.updateCategory:
        return services.alerts.actionError(`Failed to update Category`, 'categories', response);

      case CategoriesService.deleteCategory:
        return services.alerts.actionError(`Failed to delete Category`, 'categories', response);
    }
  },
});

import { useCallback } from 'react';
import { TableRow, TableColumn, TableCell } from '@heroui/react';;
import { RequestBrokerDataTable } from '@livefree/react-ui';

const columns = [
  { propertyName: 'id', name: 'Id', isSortable: true },
];

export const CategoriesList = ({ categories = [], filters, onSelect }) => {
  const renderCell = useCallback((row, columnKey) => {
    const cellValue = row[columnKey];

    switch (columnKey) {
      default:
        return cellValue;
    }
  }, []);

  return (
    <RequestBrokerDataTable
      columns={columns}
      items={ categories }
      filters={filters}
      renderColumn={column => (
        <TableColumn
          key={column.propertyName}
          allowsSorting={column?.isSortable}
        >
          {column.name}
        </TableColumn>
      )}
      renderRow={row => (
        <TableRow key={row.id} onClick={() => onSelect(row)}>
          {columnKey => <TableCell key={columnKey}>{renderCell(row, columnKey)}</TableCell>}
        </TableRow>
      )}
      withPagination
      withSorting
      withColumnSelection
    />
  );
};

  features/categories/controllers/CategoriesListController.jsx:
import { useCallback } from 'react';
import { Reactive } from '@livefree/reactive';
import { RequestBroker, queryDexieWithMetadata } from '@livefree/applications';
import { CategoriesList } from 'features/categories/components';

export const CategoriesListController = Reactive(
  ({ data, services, monitors, db, onSelect }) => {
    const { categories } = data;

    const handleExecute = useCallback((action, metadata) => {
      action(metadata);
    }, []);

    return (
      <RequestBroker
        actionKey="queryCategories"
        action={services.categories.queryCategories}
        defaultSorting={ { propertyName: 'id', sortType: false } }
        defaultPaging={ { skip: 0, take: 50, page: 1, totalCount: 0 } }
        isExecuting={monitors.queryCategories.executing}
        skipInitial={() => db.categories.toCollection().count()}
        onExecute={handleExecute}
      >
        <CategoriesList
          categories={ categories }
          isExecuting={monitors.queryCategories.executing}
          onSelect={item => onSelect && onSelect(item)}
        />
      </RequestBroker>
    );
  },
  {
    name: 'CategoriesListController',
    queries: (_, services) => [
      {
        name: 'categories',
        execute: db =>
          queryDexieWithMetadata(db, services.categories.queryCategories.action, () => db.categories, {
            sorting: true,
          }),
        deps: [],
      },
    ],
    monitors: services => [services.categories.queryCategories],
  },
);

  features/categories/controllers/CategoryUpdateController.jsx:
import { useState } from 'react';
import { Reactive, useReactorOnSuccess } from '@livefree/reactive';
import { DeleteButton, SaveButton, withModalController } from '@livefree/react-ui';
import { CategoryForm } from 'features/categories/components';

const CategoryUpdateController = Reactive(
  ({ services, data, monitors, categoryId, onSuccess, onClose }) => {
    const { category } = data;
    const [form, setForm] = useState({});

    useReactorOnSuccess(services.categories, 'updateCategory', payload => {
      onSuccess && onSuccess(payload);
    });

    useReactorOnSuccess(services.categories, 'deleteCategory', () => {
      onClose && onClose();
    });

    return (
      <div className="flex flex-col gap-2">
        <CategoryForm category={ category } onChange={setForm} />
        <div className="flex justify-between gap-2">
          <DeleteButton
            size="sm"
            onPress={() => services.categories.deleteCategory(categoryId)}
            isLoading={monitors.deleteCategory.executing}
          />
          <SaveButton
            size="sm"
            isDisabled={!form.isValid}
            onPress={() =>
              services.categories.updateCategory(categoryId, form.data)
            }
            isLoading={monitors.updateCategory.executing}
          />
        </div>
      </div>
    );
  },
  {
    name: 'CategoryUpdateController',
    sync: (services, { categoryId }) => [
      {
        execute: () => services.categories.getCategory(categoryId),
        deps: [categoryId],
      },
    ],
    queries: ({ categoryId }) => [
      {
        name: 'category',
        defaultValue: {},
        execute: db => db.categories.get(categoryId),
        deps: [categoryId],
      },
    ],
    monitors: services => [
      services.categories.updateCategory,
      services.categories.deleteCategory,
    ],
  },
);

CategoryUpdateController.Modal = withModalController(({ onSuccess, ...props }) => (
  <CategoryUpdateController onSuccess={onSuccess} {...props} />
));

export { CategoryUpdateController };

{
    public static void MapCategoriesRoutes(
        this WebApplication app,

        string prefix = "categories")

        var group = app
            .MapModelRoutes<CategoryModel, ICategoriesEndpoint, long>(basePrefix, prefix, "Categories")

            .AddUpdate()
            .AddDelete()
            .Builder
