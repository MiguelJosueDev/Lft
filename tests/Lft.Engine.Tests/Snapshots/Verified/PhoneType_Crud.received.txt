import { ReactiveService } from '@livefree/reactive';
import { GeneratedSDK } from 'sdk/GeneratedSDK';
import { GeneratedConfig } from 'core/config';
const PhoneTypesClient = GeneratedSDK.PhoneTypesClient;

export const PhoneTypesService = new (ReactiveService('phoneTypes', PhoneTypesClient, {
  recursive: true,
  recurseDepth: 2,
}))(GeneratedConfig);

export const PhoneTypesReactor = PhoneTypesService.reactor('phoneTypesReactor', {
  success: (response, services, db) => {
    const { source, args, payload } = response;

    switch (source) {

      case PhoneTypesService.getPhoneType:
        return db.phoneTypes.put(payload);

      case PhoneTypesService.createPhoneType:
        services.alerts.actionSuccess(`PhoneType added successfully`, 'phoneTypes', response);
        return db.phoneTypes.add(payload);

      case PhoneTypesService.updatePhoneType:
        services.alerts.actionSuccess(`PhoneType updated successfully`, 'phoneTypes', response);
        return db.phoneTypes.put(payload);

      case PhoneTypesService.deletePhoneType:
        services.alerts.actionSuccess(`PhoneType deleted successfully`, 'phoneTypes', response);
        return db.phoneTypes.delete(args[0]);
    }
  },
  error: (response, services) => {
    switch (response.source) {

      case PhoneTypesService.getPhoneType:
        return services.alerts.actionError(`Failed to fetch PhoneType details`, 'phoneTypes', response);

      case PhoneTypesService.createPhoneType:
        return services.alerts.actionError(`Failed to create PhoneType`, 'phoneTypes', response);

      case PhoneTypesService.updatePhoneType:
        return services.alerts.actionError(`Failed to update PhoneType`, 'phoneTypes', response);

      case PhoneTypesService.deletePhoneType:
        return services.alerts.actionError(`Failed to delete PhoneType`, 'phoneTypes', response);
    }
  },
});

import { useCallback } from 'react';
import { TableRow, TableColumn, TableCell } from '@heroui/react';;
import { RequestBrokerDataTable } from '@livefree/react-ui';

const columns = [
  { propertyName: 'id', name: 'Id', isSortable: true },
];

export const PhoneTypesList = ({ phoneTypes = [], filters, onSelect }) => {
  const renderCell = useCallback((row, columnKey) => {
    const cellValue = row[columnKey];

    switch (columnKey) {
      default:
        return cellValue;
    }
  }, []);

  return (
    <RequestBrokerDataTable
      columns={columns}
      items={ phoneTypes }
      filters={filters}
      renderColumn={column => (
        <TableColumn
          key={column.propertyName}
          allowsSorting={column?.isSortable}
        >
          {column.name}
        </TableColumn>
      )}
      renderRow={row => (
        <TableRow key={row.id} onClick={() => onSelect(row)}>
          {columnKey => <TableCell key={columnKey}>{renderCell(row, columnKey)}</TableCell>}
        </TableRow>
      )}
      withPagination
      withSorting
      withColumnSelection
    />
  );
};

import { useCallback } from 'react';
import { Reactive } from '@livefree/reactive';
import { RequestBroker, queryDexieWithMetadata } from '@livefree/applications';
import { PhoneTypesList } from 'features/phoneTypes/components';

export const PhoneTypesListController = Reactive(
  ({ data, services, monitors, db, onSelect }) => {
    const { phoneTypes } = data;

    const handleExecute = useCallback((action, metadata) => {
      action(metadata);
    }, []);

    return (
      <RequestBroker
        actionKey="queryPhoneTypes"
        action={services.phoneTypes.queryPhoneTypes}
        defaultSorting={ { propertyName: 'id', sortType: false } }
        defaultPaging={ { skip: 0, take: 50, page: 1, totalCount: 0 } }
        isExecuting={monitors.queryPhoneTypes.executing}
        skipInitial={() => db.phoneTypes.toCollection().count()}
        onExecute={handleExecute}
      >
        <PhoneTypesList
          phoneTypes={ phoneTypes }
          isExecuting={monitors.queryPhoneTypes.executing}
          onSelect={item => onSelect && onSelect(item)}
        />
      </RequestBroker>
    );
  },
  {
    name: 'PhoneTypesListController',
    queries: (_, services) => [
      {
        name: 'phoneTypes',
        execute: db =>
          queryDexieWithMetadata(db, services.phoneTypes.queryPhoneTypes.action, () => db.phoneTypes, {
            sorting: true,
          }),
        deps: [],
      },
    ],
    monitors: services => [services.phoneTypes.queryPhoneTypes],
  },

import { useState } from 'react';
import { Reactive, useReactorOnSuccess } from '@livefree/reactive';
import { DeleteButton, SaveButton, withModalController } from '@livefree/react-ui';
import { PhoneTypeForm } from 'features/phoneTypes/components';

const PhoneTypeUpdateController = Reactive(
  ({ services, data, monitors, phoneTypeId, onSuccess, onClose }) => {
    const { phoneType } = data;
    const [form, setForm] = useState({});

    useReactorOnSuccess(services.phoneTypes, 'updatePhoneType', payload => {
      onSuccess && onSuccess(payload);
    });

    useReactorOnSuccess(services.phoneTypes, 'deletePhoneType', () => {
      onClose && onClose();
    });

    return (
      <div className="flex flex-col gap-2">
        <PhoneTypeForm phoneType={ phoneType } onChange={setForm} />
        <div className="flex justify-between gap-2">
          <DeleteButton
            size="sm"
            onPress={() => services.phoneTypes.deletePhoneType(phoneTypeId)}
            isLoading={monitors.deletePhoneType.executing}
          />
          <SaveButton
            size="sm"
            isDisabled={!form.isValid}
            onPress={() =>
              services.phoneTypes.updatePhoneType(phoneTypeId, form.data)
            }
            isLoading={monitors.updatePhoneType.executing}
          />
        </div>
      </div>
    );
  },
  {
    name: 'PhoneTypeUpdateController',
    sync: (services, { phoneTypeId }) => [
      {
        execute: () => services.phoneTypes.getPhoneType(phoneTypeId),
        deps: [phoneTypeId],
      },
    ],
    queries: ({ phoneTypeId }) => [
      {
        name: 'phoneType',
        defaultValue: {},
        execute: db => db.phoneTypes.get(phoneTypeId),
        deps: [phoneTypeId],
      },
    ],
    monitors: services => [
      services.phoneTypes.updatePhoneType,
      services.phoneTypes.deletePhoneType,
    ],
  },
);

PhoneTypeUpdateController.Modal = withModalController(({ onSuccess, ...props }) => (
  <PhoneTypeUpdateController onSuccess={onSuccess} {...props} />
));

export { PhoneTypeUpdateController };
