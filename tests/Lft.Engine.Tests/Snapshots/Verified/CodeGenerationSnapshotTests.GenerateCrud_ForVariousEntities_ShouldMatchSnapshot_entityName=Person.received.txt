import { ReactiveService } from '@livefree/reactive';
import { GeneratedSDK } from 'sdk/GeneratedSDK';
import { GeneratedConfig } from 'core/config';
const PeopleClient = GeneratedSDK.PeopleClient;

export const PeopleService = new (ReactiveService('people', PeopleClient, {
  recursive: true,
  recurseDepth: 2,
}))(GeneratedConfig);

export const PeopleReactor = PeopleService.reactor('peopleReactor', {
  success: (response, services, db) => {
    const { source, args, payload } = response;

    switch (source) {

      case PeopleService.getPerson:
        return db.people.put(payload);

      case PeopleService.createPerson:
        services.alerts.actionSuccess(`Person added successfully`, 'people', response);
        return db.people.add(payload);

      case PeopleService.updatePerson:
        services.alerts.actionSuccess(`Person updated successfully`, 'people', response);
        return db.people.put(payload);

      case PeopleService.deletePerson:
        services.alerts.actionSuccess(`Person deleted successfully`, 'people', response);
        return db.people.delete(args[0]);
    }
  },
  error: (response, services) => {
    switch (response.source) {

      case PeopleService.getPerson:
        return services.alerts.actionError(`Failed to fetch Person details`, 'people', response);

      case PeopleService.createPerson:
        return services.alerts.actionError(`Failed to create Person`, 'people', response);

      case PeopleService.updatePerson:
        return services.alerts.actionError(`Failed to update Person`, 'people', response);

      case PeopleService.deletePerson:
        return services.alerts.actionError(`Failed to delete Person`, 'people', response);
    }
  },
});

  features/people/components/PeopleList.jsx:
import { useCallback } from 'react';
import { TableRow, TableColumn, TableCell } from '@heroui/react';;
import { RequestBrokerDataTable } from '@livefree/react-ui';

const columns = [
  { propertyName: 'id', name: 'Id', isSortable: true },
];

export const PeopleList = ({ people = [], filters, onSelect }) => {
  const renderCell = useCallback((row, columnKey) => {
    const cellValue = row[columnKey];

    switch (columnKey) {
      default:
        return cellValue;
    }
  }, []);

  return (
    <RequestBrokerDataTable
      columns={columns}
      items={ people }
      filters={filters}
      renderColumn={column => (
        <TableColumn
          key={column.propertyName}
          allowsSorting={column?.isSortable}
        >
          {column.name}
        </TableColumn>
      )}
      renderRow={row => (
        <TableRow key={row.id} onClick={() => onSelect(row)}>
          {columnKey => <TableCell key={columnKey}>{renderCell(row, columnKey)}</TableCell>}
        </TableRow>
      )}
      withPagination
      withSorting
      withColumnSelection
    />
  );
};

import { useCallback } from 'react';
import { Reactive } from '@livefree/reactive';
import { RequestBroker, queryDexieWithMetadata } from '@livefree/applications';
import { PeopleList } from 'features/people/components';

export const PeopleListController = Reactive(
  ({ data, services, monitors, db, onSelect }) => {
    const { people } = data;

    const handleExecute = useCallback((action, metadata) => {
      action(metadata);
    }, []);

    return (
      <RequestBroker
        actionKey="queryPeople"
        action={services.people.queryPeople}
        defaultSorting={ { propertyName: 'id', sortType: false } }
        defaultPaging={ { skip: 0, take: 50, page: 1, totalCount: 0 } }
        isExecuting={monitors.queryPeople.executing}
        skipInitial={() => db.people.toCollection().count()}
        onExecute={handleExecute}
      >
        <PeopleList
          people={ people }
          isExecuting={monitors.queryPeople.executing}
          onSelect={item => onSelect && onSelect(item)}
        />
      </RequestBroker>
    );
  },
  {
    name: 'PeopleListController',
    queries: (_, services) => [
      {
        name: 'people',
        execute: db =>
          queryDexieWithMetadata(db, services.people.queryPeople.action, () => db.people, {
            sorting: true,
          }),
        deps: [],
      },
    ],
    monitors: services => [services.people.queryPeople],
  },
);

import { useState } from 'react';
import { Reactive, useReactorOnSuccess } from '@livefree/reactive';
import { DeleteButton, SaveButton, withModalController } from '@livefree/react-ui';
import { PersonForm } from 'features/people/components';

const PersonUpdateController = Reactive(
  ({ services, data, monitors, personId, onSuccess, onClose }) => {
    const { person } = data;
    const [form, setForm] = useState({});

    useReactorOnSuccess(services.people, 'updatePerson', payload => {
      onSuccess && onSuccess(payload);
    });

    useReactorOnSuccess(services.people, 'deletePerson', () => {
      onClose && onClose();
    });

    return (
      <div className="flex flex-col gap-2">
        <PersonForm person={ person } onChange={setForm} />
        <div className="flex justify-between gap-2">
          <DeleteButton
            size="sm"
            onPress={() => services.people.deletePerson(personId)}
            isLoading={monitors.deletePerson.executing}
          />
          <SaveButton
            size="sm"
            isDisabled={!form.isValid}
            onPress={() =>
              services.people.updatePerson(personId, form.data)
            }
            isLoading={monitors.updatePerson.executing}
          />
        </div>
      </div>
    );
  },
  {
    name: 'PersonUpdateController',
    sync: (services, { personId }) => [
      {
        execute: () => services.people.getPerson(personId),
        deps: [personId],
      },
    ],
    queries: ({ personId }) => [
      {
        name: 'person',
        defaultValue: {},
        execute: db => db.people.get(personId),
        deps: [personId],
      },
    ],
    monitors: services => [
      services.people.updatePerson,
      services.people.deletePerson,
    ],
  },
);

PersonUpdateController.Modal = withModalController(({ onSuccess, ...props }) => (
  <PersonUpdateController onSuccess={onSuccess} {...props} />
));

export { PersonUpdateController };
