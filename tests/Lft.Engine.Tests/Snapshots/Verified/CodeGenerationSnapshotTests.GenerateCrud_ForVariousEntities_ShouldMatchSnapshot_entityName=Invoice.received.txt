import { ReactiveService } from '@livefree/reactive';
import { GeneratedSDK } from 'sdk/GeneratedSDK';
import { GeneratedConfig } from 'core/config';
const InvoicesClient = GeneratedSDK.InvoicesClient;

export const InvoicesService = new (ReactiveService('invoices', InvoicesClient, {
  recursive: true,
  recurseDepth: 2,
}))(GeneratedConfig);

export const InvoicesReactor = InvoicesService.reactor('invoicesReactor', {
  success: (response, services, db) => {
    const { source, args, payload } = response;

    switch (source) {

      case InvoicesService.getInvoice:
        return db.invoices.put(payload);

      case InvoicesService.createInvoice:
        services.alerts.actionSuccess(`Invoice added successfully`, 'invoices', response);
        return db.invoices.add(payload);

      case InvoicesService.updateInvoice:
        services.alerts.actionSuccess(`Invoice updated successfully`, 'invoices', response);
        return db.invoices.put(payload);

      case InvoicesService.deleteInvoice:
        services.alerts.actionSuccess(`Invoice deleted successfully`, 'invoices', response);
        return db.invoices.delete(args[0]);
    }
  },
  error: (response, services) => {
    switch (response.source) {

      case InvoicesService.getInvoice:
        return services.alerts.actionError(`Failed to fetch Invoice details`, 'invoices', response);

      case InvoicesService.createInvoice:
        return services.alerts.actionError(`Failed to create Invoice`, 'invoices', response);

      case InvoicesService.updateInvoice:
        return services.alerts.actionError(`Failed to update Invoice`, 'invoices', response);

      case InvoicesService.deleteInvoice:
        return services.alerts.actionError(`Failed to delete Invoice`, 'invoices', response);
    }
  },
});

import { useCallback } from 'react';
import { TableRow, TableColumn, TableCell } from '@heroui/react';;
import { RequestBrokerDataTable } from '@livefree/react-ui';

const columns = [
  { propertyName: 'id', name: 'Id', isSortable: true },
];

export const InvoicesList = ({ invoices = [], filters, onSelect }) => {
  const renderCell = useCallback((row, columnKey) => {
    const cellValue = row[columnKey];

    switch (columnKey) {
      default:
        return cellValue;
    }
  }, []);

  return (
    <RequestBrokerDataTable
      columns={columns}
      items={ invoices }
      filters={filters}
      renderColumn={column => (
        <TableColumn
          key={column.propertyName}
          allowsSorting={column?.isSortable}
        >
          {column.name}
        </TableColumn>
      )}
      renderRow={row => (
        <TableRow key={row.id} onClick={() => onSelect(row)}>
          {columnKey => <TableCell key={columnKey}>{renderCell(row, columnKey)}</TableCell>}
        </TableRow>
      )}
      withPagination
      withSorting
      withColumnSelection
    />
  );
};

import { useCallback } from 'react';
import { Reactive } from '@livefree/reactive';
import { RequestBroker, queryDexieWithMetadata } from '@livefree/applications';
import { InvoicesList } from 'features/invoices/components';

export const InvoicesListController = Reactive(
  ({ data, services, monitors, db, onSelect }) => {
    const { invoices } = data;

    const handleExecute = useCallback((action, metadata) => {
      action(metadata);
    }, []);

    return (
      <RequestBroker
        actionKey="queryInvoices"
        action={services.invoices.queryInvoices}
        defaultSorting={ { propertyName: 'id', sortType: false } }
        defaultPaging={ { skip: 0, take: 50, page: 1, totalCount: 0 } }
        isExecuting={monitors.queryInvoices.executing}
        skipInitial={() => db.invoices.toCollection().count()}
        onExecute={handleExecute}
      >
        <InvoicesList
          invoices={ invoices }
          isExecuting={monitors.queryInvoices.executing}
          onSelect={item => onSelect && onSelect(item)}
        />
      </RequestBroker>
    );
  },
  {
    name: 'InvoicesListController',
    queries: (_, services) => [
      {
        name: 'invoices',
        execute: db =>
          queryDexieWithMetadata(db, services.invoices.queryInvoices.action, () => db.invoices, {
            sorting: true,
          }),
        deps: [],
      },
    ],
    monitors: services => [services.invoices.queryInvoices],
  },
);

import { useState } from 'react';
import { Reactive, useReactorOnSuccess } from '@livefree/reactive';
import { DeleteButton, SaveButton, withModalController } from '@livefree/react-ui';
import { InvoiceForm } from 'features/invoices/components';

const InvoiceUpdateController = Reactive(
  ({ services, data, monitors, invoiceId, onSuccess, onClose }) => {
    const { invoice } = data;
    const [form, setForm] = useState({});

    useReactorOnSuccess(services.invoices, 'updateInvoice', payload => {
      onSuccess && onSuccess(payload);
    });

    useReactorOnSuccess(services.invoices, 'deleteInvoice', () => {
      onClose && onClose();
    });

    return (
      <div className="flex flex-col gap-2">
        <InvoiceForm invoice={ invoice } onChange={setForm} />
        <div className="flex justify-between gap-2">
          <DeleteButton
            size="sm"
            onPress={() => services.invoices.deleteInvoice(invoiceId)}
            isLoading={monitors.deleteInvoice.executing}
          />
          <SaveButton
            size="sm"
            isDisabled={!form.isValid}
            onPress={() =>
              services.invoices.updateInvoice(invoiceId, form.data)
            }
            isLoading={monitors.updateInvoice.executing}
          />
        </div>
      </div>
    );
  },
  {
    name: 'InvoiceUpdateController',
    sync: (services, { invoiceId }) => [
      {
        execute: () => services.invoices.getInvoice(invoiceId),
        deps: [invoiceId],
      },
    ],
    queries: ({ invoiceId }) => [
      {
        name: 'invoice',
        defaultValue: {},
        execute: db => db.invoices.get(invoiceId),
        deps: [invoiceId],
      },
    ],
    monitors: services => [
      services.invoices.updateInvoice,
      services.invoices.deleteInvoice,
    ],
  },
);

InvoiceUpdateController.Modal = withModalController(({ onSuccess, ...props }) => (
  <InvoiceUpdateController onSuccess={onSuccess} {...props} />
));

export { InvoiceUpdateController };
